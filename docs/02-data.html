3  数据读写与基本操作

本章介绍 R 语言中的数据导入导出、数据探索和基础可视化。

3.1 多格式数据导入导出

3.1.1 读取 Stata .dta 文件

使用 haven 包读取 Stata 文件：

# 加载包
library(haven)
library(tibble)
library(dplyr)

# 设置工作路径
setwd("F:/2026学术直播/2026年5月学术直播资料讲义/16--R 工作流")

# 读取 .dta 文件
simulation <- read_dta("simulation.dta")

# 查看数据结构
glimpse(simulation)

3.2 数据探索

3.2.1 查看数据

# 前几行
head(simulation)

# 后几行
tail(simulation)

# 数据结构
str(simulation)

# 统计摘要
summary(simulation)

3.2.2 使用 dplyr 进行数据操作

library(dplyr)

# 筛选观测
simulation %>%
  filter(t >= 2015)

# 选择变量
simulation %>%
  select(i, t, Y, D)

# 生成新变量
simulation <- simulation %>%
  mutate(post = as.integer(t >= 2015))

# 分组统计
simulation %>%
  group_by(t) %>%
  summarize(
    mean_Y = mean(Y),
    n = n()
  )

3.3 基础可视化

3.3.1 基础绘图函数

# 散点图
plot(simulation$t, simulation$Y,
     xlab = "年份", ylab = "结果变量",
     main = "Y 的时间趋势")

# 直方图
hist(simulation$Y,
     xlab = "Y", main = "Y 的分布",
     col = "lightblue", border = "white")

3.3.2 使用 ggplot2

library(ggplot2)
library(dplyr)

# 直方图
ggplot(simulation, aes(x = Y)) +
  geom_histogram(fill = "steelblue", color = "white", bins = 30) +
  labs(title = "Y 的分布", x = "Y", y = "频数") +
  theme_minimal()

# 时间趋势图
simulation %>%
  group_by(t, D) %>%
  summarize(mean_Y = mean(Y), .groups = "drop") %>%
  ggplot(aes(x = t, y = mean_Y, color = factor(D), group = D)) +
  geom_line(linewidth = 1) +
  geom_point(size = 2) +
  labs(title = "处理组与对照组趋势",
       x = "年份", y = "平均 Y", color = "处理状态") +
  scale_color_manual(labels = c("对照组", "处理组"),
                     values = c("#440154", "#21908C")) +
  theme_minimal()

3.3.3 面板数据可视化

使用 panelView 包：

library(panelView)

# 面板数据状态图
panelview(Y ~ D, data = simulation,
          index = c("i", "t"),
          xlab = "时间", ylab = "个体",
          main = "处理状态分布",
          by.timing = FALSE,
          legend.labs = c("未处理", "已处理"),
          gridOff = TRUE)
