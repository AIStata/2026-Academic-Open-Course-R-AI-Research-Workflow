8.2 综合实践项目

8.2.1 项目背景

假设你正在研究一项政策的效果，该政策在不同地区在不同时间点实施。你需要：

  1. 读取数据
  2. 进行描述性统计
  3. 估计政策效果
  4. 生成图表和表格
  5. 撰写分析报告

8.2.2 数据准备

让我们创建一个完整的模拟数据集供实践使用：

set.seed(456)
library(tibble)
library(dplyr)

# 创建模拟面板数据
n_regions <- 50
n_years <- 12

policy_data <- tibble(
  region_id = rep(1:n_regions, each = n_years),
  year = rep(2010:2021, n_regions),
  # 随机分配政策实施年份
  treat_year = sample(c(0, 2015, 2016, 2017, 2018),
                      n_regions,
                      replace = TRUE,
                      prob = c(0.25, 0.2, 0.2, 0.2, 0.15)) %>%
    rep(each = n_years),
  # 协变量
  pop = rnorm(n_regions * n_years, 100, 20) %>% abs(),
  gdp = rnorm(n_regions * n_years, 500, 100) %>% abs()
) %>%
  mutate(
    # 处理状态
    treat = as.integer(treat_year > 0 & year >= treat_year),
    # 相对时间
    rel_year = ifelse(treat_year == 0, Inf, year - treat_year),
    # 结果变量（包含政策效应）
    outcome = 10 +
      0.1 * gdp +
      0.05 * pop +
      (treat_year > 0) * 2 +
      treat * (3 + 0.5 * pmax(rel_year, 0)) +
      rnorm(n())
  )

head(policy_data)

8.2.4 任务 2：可视化处理状态

library(panelView)

# 面板视图
panelview(outcome ~ treat,
          data = policy_data,
          index = c("region_id", "year"),
          xlab = "年份", ylab = "地区",
          main = "政策实施状态",
          by.timing = TRUE)

8.2.5 任务 3：估计政策效果

library(did)

# 估计 ATT(g,t)
attgt_result <- att_gt(
  yname = "outcome",
  gname = "treat_year",
  idname = "region_id",
  tname = "year",
  xformla = ~ gdp + pop,
  data = policy_data,
  est_method = "dr"
)

# 查看结果
summary(attgt_result)

8.2.6 任务 4：事件研究分析

# 动态效应
es_result <- aggte(attgt_result, type = "dynamic")

# 绘制事件研究图
ggdid(es_result) +
  labs(title = "事件研究：政策动态效应",
       x = "相对时间", y = "ATT") +
  theme_minimal()

8.2.7 任务 5：生成回归表格

library(modelsummary)
library(fixest)

# 基础 TWFE 估计
twfe_model <- feols(outcome ~ treat | region_id + year,
                    data = policy_data)

# 加入协变量
twfe_model_cov <- feols(outcome ~ treat + gdp + pop | region_id + year,
                        data = policy_data)

# 输出表格
modelsummary(
  list("TWFE (无协变量)" = twfe_model,
       "TWFE (有协变量)" = twfe_model_cov),
  output = "table_regression.docx",
  stars = TRUE,
  gof_map = c("nobs", "r.squared", "F")
)

8.4 研究工作流建议

8.4.1 项目组织

my-research/
├── data/
│   ├── raw/           # 原始数据
│   └── processed/     # 处理后数据
├── code/
│   ├── 01-clean-data.R
│   ├── 02-analysis.R
│   └── 03-figures.R
├── output/
│   ├── figures/
│   └── tables/
└── paper/
    ├── draft.qmd
    └── references.bib
